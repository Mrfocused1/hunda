-- ============================================================
-- Supabase RLS Policy Fix for Admin Operations
-- Run this in the Supabase SQL Editor
-- ============================================================

-- 1. Enable RLS on products table (if not already enabled)
ALTER TABLE products ENABLE ROW LEVEL SECURITY;

-- 2. Drop existing policies if they exist (to avoid conflicts)
DROP POLICY IF EXISTS "Allow anonymous read" ON products;
DROP POLICY IF EXISTS "Allow anonymous insert" ON products;
DROP POLICY IF EXISTS "Allow anonymous update" ON products;
DROP POLICY IF EXISTS "Allow anonymous delete" ON products;

-- 3. Create new policies for admin operations
-- Allow anyone to read products (for the storefront)
CREATE POLICY "Allow anonymous read" ON products
    FOR SELECT USING (true);

-- Allow anyone to insert products (for admin - in production, use proper auth)
CREATE POLICY "Allow anonymous insert" ON products
    FOR INSERT WITH CHECK (true);

-- Allow anyone to update products (for admin - in production, use proper auth)
CREATE POLICY "Allow anonymous update" ON products
    FOR UPDATE USING (true);

-- Allow anyone to delete products (for admin - in production, use proper auth)
CREATE POLICY "Allow anonymous delete" ON products
    FOR DELETE USING (true);

-- ============================================================
-- Storage Bucket RLS Policies
-- ============================================================

-- 1. Enable RLS on storage objects (if not already enabled)
ALTER TABLE storage.objects ENABLE ROW LEVEL SECURITY;

-- 2. Drop existing storage policies for product-images bucket
DROP POLICY IF EXISTS "Allow anonymous read product-images" ON storage.objects;
DROP POLICY IF EXISTS "Allow anonymous insert product-images" ON storage.objects;
DROP POLICY IF EXISTS "Allow anonymous update product-images" ON storage.objects;
DROP POLICY IF EXISTS "Allow anonymous delete product-images" ON storage.objects;

-- 3. Create storage policies for product-images bucket
-- Allow anyone to read images
CREATE POLICY "Allow anonymous read product-images" ON storage.objects
    FOR SELECT USING (bucket_id = 'product-images');

-- Allow anyone to upload images
CREATE POLICY "Allow anonymous insert product-images" ON storage.objects
    FOR INSERT WITH CHECK (bucket_id = 'product-images');

-- Allow anyone to update images
CREATE POLICY "Allow anonymous update product-images" ON storage.objects
    FOR UPDATE USING (bucket_id = 'product-images');

-- Allow anyone to delete images
CREATE POLICY "Allow anonymous delete product-images" ON storage.objects
    FOR DELETE USING (bucket_id = 'product-images');
